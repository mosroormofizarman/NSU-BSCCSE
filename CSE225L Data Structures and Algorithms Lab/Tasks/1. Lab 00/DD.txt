 #include <iostream> using namespace std; class Box {     // instance members private:     double height, width, depth;  public:     // constructor     Box(double height, double width, double depth)     {         this->height = height;         this->width = width;         this->depth = depth;     }     // reutnrs the volume     double volume()     {         return height * width * depth;     } }; int main() {     // creating the Box objects and displaying the volume     Box b1(10, 20, 30), b2(10, 22, 32);     cout << "volume of Box(10,20,30) : " << b1.volume() << endl;     cout << "volume of Box(100,200,300) : " << b2.volume() << endl;     return 0; }
